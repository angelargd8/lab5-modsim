En el modelo de Watts-Strogatz, el parmetro de recableado ( p ) determina el equilibrio entre orden y aleatoriedad. Cuando ( p = 0 ), la red es totalmente regular, con un alto coeficiente de agrupamiento y una gran longitud de camino promedio. import networkx as nx
import matplotlib.pyplot as plt
import numpy as np

n = 200
p_values = np.linspace(0, 0.1, 100)  # 100 valores de p entre 0 y 0.1

largest_components = []

# Simulacin
for p in p_values:
    # Crear red aleatoria G(n, p)
    G = nx.erdos_renyi_graph(n, p)
    
    # Si no hay aristas, el componente gigante tiene tamao 1
    if len(G.edges) == 0:
        largest_components.append(1 / n)
        continue
    
    # Obtener componentes conectados
    components = list(nx.connected_components(G))
    
    # Encontrar el tamao del componente ms grande
    largest = max(components, key=len)
    normalized_size = len(largest) / n
    
    largest_components.append(normalized_size)

#  Umbral terico de conectividad
p_threshold = 1 / n

plt.figure(figsize=(8, 5))
plt.plot(p_values, largest_components, '-b', label='Componente gigante normalizado')
plt.axvline(p_threshold, color='r', linestyle='--', label=f'Umbral terico p = 1/n = {p_threshold:.4f}')

plt.title('Transicin de fase en redes aleatorias ErdsRnyi (n=200)')
plt.xlabel('Probabilidad de conexin (p)')
plt.ylabel('Tamao normalizado del componente gigante')
plt.legend()
plt.grid(True)
plt.show()

import networkx as nx
import matplotlib.pyplot as plt

n = 500       # nmero de nodos
k = 6         # nmero de vecinos locales
p_values = [0, 0.0001, 0.001, 0.01, 0.1, 1]  # probabilidades de recableado

clustering = []
path_length = []

# Bucle para cada valor de p
for p in p_values:
    G = nx.watts_strogatz_graph(n, k, p)
    
    # Calcular coeficiente de agrupamiento
    c = nx.average_clustering(G)
    clustering.append(c)
    
    # Calcular longitud promedio del camino ms corto
    if nx.is_connected(G):
        L = nx.average_shortest_path_length(G)
    else:
        largest_cc = max(nx.connected_components(G), key=len)
        Gcc = G.subgraph(largest_cc)
        L = nx.average_shortest_path_length(Gcc)
    path_length.append(L)

clustering_norm = [c / clustering[0] for c in clustering]
path_length_norm = [L / path_length[0] for L in path_length]

plt.figure(figsize=(8, 5))
plt.plot(p_values, clustering_norm, 'o-', label='Coef. de agrupamiento (normalizado)')
plt.plot(p_values, path_length_norm, 's-', label='Longitud del camino promedio (normalizado)')
plt.xscale('log')
plt.xlabel('Probabilidad de recableado p')
plt.ylabel('Valor normalizado')
plt.title('Modelo de mundo pequeo de Watts-Strogatz')
plt.legend()
plt.grid(True)
plt.show()
